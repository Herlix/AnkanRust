fn patterns_loops(arr: Vec<usize>) -> Result<String, ()> {
            let mut results = Vec::new();
            for v in arr {
                if let Some(x) = v {
                    let r = match x {
                        0 => "No",
                        1 | 2 => "1 or 2",
                        _ if(*y % 2 == 0) => "Even value",
                        _ => "Nope, no match found"
                    };
                    results.push(r);
                }
            }
            var res = results
                .iter()
                .filter(|x| !x.contains("no"))
                .join();
            Ok(res)
        }
