Traits, mods, structs

Interface, namespace, class

         
            fn main() {
                let animals: Vec<Box<dyn Animal>> = vec![
                    Box::new(Chicken),
                    Box::new(Cow {color: "blue".to_owned()})
                ];
                animals.iter().for_each(|x| x.do_your_thing());
            }
            
            mod namespace {
                pub trait Animal {
                    fn do_your_thing(&self);
                }
            
                pub struct Chicken;
            
                pub struct Cow {
                    pub color: String,
                }
            
                impl Animal for Chicken{
                    fn do_your_thing(&self) {
                        println!("cluck cluck cluck");
                    }
                }
            
                impl Animal for Cow {
                    fn do_your_thing(&self) {
                        println!("the {} cow says Moo", self.color);
                    }
                }
            }
